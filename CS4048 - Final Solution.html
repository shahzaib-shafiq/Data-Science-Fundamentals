import pandas as pd
cols = ["name", "sex", "births", "year"]
base = []
for year in range(1980, 2011):
    path = f"./babynames_/yob{year}.txt"
    #print(path)
    df = pd.read_csv(path, names=cols)
    df["year"] = year
    base.append(df)
data = pd.concat(base, ignore_index=True)
